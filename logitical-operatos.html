Logical Operators

Logical (&&) Operators

document.write((3<5) && (3<4)); //true

document.write((3<5) && (3<2)); //false

document.write((6<5) && (3<2)); //false

document.write((6<8) && (3<2)); //false

Logical (||) Operators

document.write((3<5) || (3<4)); //true

document.write((3<5) || (3<2)); //true

document.write((6<5) || (3<2)); //false

document.write((6<8) || (3<2)); //true

Logical (!) Operators

document.write(!(3>5)); //true

document.write(!(3<5)); //false

document.write(!(6<5)); //true

document.write(!(6<8)); //true
